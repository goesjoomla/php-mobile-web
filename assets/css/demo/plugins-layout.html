IE6 fallback

You might already know that IE6 does not support the :hover dynamic pseudo-class selector for any HTML element other than <a>. This problem causes our hover menu and Suckerfish dropdown menu never working as expected in IE6 without a proper fix. Below is the Javascript fallback to make these menus working like a champ in IE6. You simply need to put the following code snippet right before the </body> tag.

<script type="text/javascript"> for (var t in {ul: '', dl: ''}) { var lists = document.getElementsByTagName(t); for (var i = 0; i < lists.length; i++) { if (lists[i].className.match(/\bhover-menu\b/)) { lists[i].onmouseover = function() { this.className += ' mouseover'; }; lists[i].onmouseout = function() { this.className = this.className.replace(' mouseover', ''); }; } else if (lists[i].className.match(/\bsf-menu\b/)) { var items = lists[i].getElementsByTagName('li'); for (var j = 0; j < items.length; j++) { items[j].onmouseover = function() { this.className += ' mouseover'; }; items[j].onmouseout = function() { this.className = this.className.replace(' mouseover', ''); }; } } } } </script>
